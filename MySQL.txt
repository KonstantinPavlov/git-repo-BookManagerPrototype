# git-repo-BookManagerPrototype

Запросы для создания базы для хранения 

CREATE DATABASE BookManager;

CREATE TABLE books(
  id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
  name VARCHAR(50) NOT NULL,
  description VARCHAR(300) NOT NULL,
  author VARCHAR(50)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

  CREATE TABLE images(
    image_id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    name VARCHAR(0),
    image BLOB,
    id INT NOT NULL ,
    CONSTRAINT books_id_fk FOREIGN KEY (id) REFERENCES books (id)

  )
